# 思路一：排序后用双指针。
```c
class Solution {
public:
    int majorityElement(vector<int>& nums) {
        int n=nums.size();
        sort(nums.begin(),nums.end());
        int res;
        int left=0;
        int right=1;
        if(n==1)
            return nums[0];
        while(right<n)
        {
            if(nums[right]>nums[left])
            {
                if((right-left)>=(n/2+1))
                    return nums[left];
                else{
                    left=right;
                }
            }
            if(right==n-1 && (right-left+1)>=(n/2+1))
                return nums[left];
            right=right+1;
        }
        return 0;
    }
};
```
