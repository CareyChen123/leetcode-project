# 思路一：遍历每一个子串，判断是否是合法的，是的话就统计长度，最后保留最长的。（超时无法通过所有测试用例）
```c
class Solution {
public:
    int longestValidParentheses(string s) {
        int res=0;
        int len=0;
        for(int i=0; i<s.size(); i++)
        {
            if(s[i]==')')
                continue;
            for(int j=i+1; j<s.size(); j=j+2)
            {
                len=j-i+1;
                // cout<<i<<" "<<j<<" "<<s.substr(i,len)<<endl;
                if(isValid(s.substr(i,len)))
                {
                    // i=i+len-1;
                    if(len>res)
                        res=len;
                    // cout<<res<<endl;
                }
            }
        }
        return res;
    }
    bool isValid(string s)
    {
        vector<char> statck;
        int top=0;
        for(int i=0; i<s.size(); i++)
        {
            // cout<<"size:"<<statck.size()<<top<<" "<<s[i]<<endl;
            if(s[i]=='(')
            {
                // cout<<"carey:"<<s[i]<<endl;
                statck.push_back(s[i]);
                top++;
            }else{
                if(top==0 || (s[i]==')'&&statck[top-1]!='('))
                {
                    // cout<<"wangle"<<statck[top-1]<<endl;
                    return false;
                }else{
                    // cout<<"carey"<<endl;
                    statck.pop_back();
                    top--;
                }
            }  
        }
        // cout<<"top:"<<top<<s<<endl;
        return top==0;
    }
};
```
# 思路二：遍历字串的过程中用栈判断合法子串长度。如果是‘（’就push下标到栈中，如果是‘）’就弹出栈顶，此时若栈为空，就把此位置下标push（代表第一个不能被匹配的右括号的位置）；若此时栈不为空，就统计以该位置结尾的字串的最大有效的串的长度i-stk.top()。
```c
class Solution {
public:
    int longestValidParentheses(string s) {
        vector<int> statck;
        statck.push_back(-1);
        int res=0;
        for(int i=0; i<s.size(); i++)
        {
            if(s[i]=='(')
                statck.push_back(i);
            else
            {
                statck.pop_back();
                if(statck.size()==0)
                {
                    // cout<<"xxx:"<<i<<endl;
                    statck.push_back(i);
                }else{
                    // cout<<i<<" "<<statck[0]<<endl;
                    res=max(res, i-statck[statck.size()-1]);
                }
            }
        }
        return res;
    }
};
```
