# 思路二：分情况讨论。是否空格，是否+或-,是否数字，还有其他。INT_MIN和INT_MAX使用。
```c
class Solution {
public:
    int myAtoi(string s) {
        int flag=1;
        int no_space_flag=0;

        int num=0;
        for(char &c : s)
        {
            if(isspace(c))
            {
                if(no_space_flag==0)
                    continue;
                else
                    break;
            }
            if(c=='-')
            {
                if(no_space_flag==0)
                {
                    no_space_flag=1;
                    flag=0;
                    continue;
                }
                else
                    break;
            }
            if(c=='+')
            {
                if(no_space_flag==0)
                {
                    no_space_flag=1;
                    continue;
                }
                else
                    break;
            }
            if(isdigit(c))
            {
                if(flag==0 && ((-num)<INT_MIN/10 || ((-num)==INT_MIN/10 && (c-'0')>=8)))
                    return INT_MIN;
                if(num>INT_MAX/10 || (num==INT_MAX/10 && (c-'0')>7))
                {
                    return INT_MAX;
                }
                no_space_flag=1;
                num=num*10+(c-'0');
                continue;
            }
            no_space_flag=1;
            break;
        }
         flag==0 ? num=-num:num=num;
         return num;   
    }
};
```
