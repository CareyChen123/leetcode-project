# 思路：二分法找位置，注意端点值处理。如果target比最大值大或小于等于最小值最特殊处理。剩下的都会在最后剩下的区间中。
```c
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int left = 0;
        int right = nums.size()-1;
        if(target > nums[right])
            return (right+1);
        if(target <= nums[left])
            return 0;
        for(;right - left > 1;)
        {
            int flag =left + (right-left)/2;
            if(nums[flag] == target)
                return flag;
            else if(nums[flag] > target)
            {
                right = flag;
            }else{
                left = flag;
            }
        }
        return (left+1);
    }
};
```
