# 思路 遍历字符串，将每个值对应的数字累加；特殊组合需要减去对应的数字。
```c
int romanToInt(char * s){
    int sum = 0;
    for(int i=0; i<strlen(s); i++)
    {
        if(s[i] == 'I')
        {
            if(i+1<strlen(s) && (s[i+1] == 'V' || s[i+1] == 'X')) // I后面紧跟V或X,且i+1不能越界
            {
                sum = sum - 1;
            }
            else
            {
                sum = sum + 1;
            }
        }
            
        if(s[i] == 'V')
            sum = sum + 5;

        if(s[i] == 'X')
        {
            if(i+1<strlen(s) && (s[i+1] == 'L' || s[i+1] == 'C'))
            {
                sum = sum - 10;
            }
            else
            {
                sum = sum + 10;
            }
        }

        if(s[i] == 'L')
            sum = sum + 50;

        if(s[i] == 'C')
        {
            if(i+1<strlen(s) && (s[i+1] == 'D' || s[i+1] == 'M'))
            {
                sum = sum - 100;
            }
            else
            {
                sum = sum + 100;
            }
        }

        if(s[i] == 'D')
            sum = sum + 500;

        if(s[i] == 'M')
            sum = sum + 1000;
    }
    return sum;

}
```
